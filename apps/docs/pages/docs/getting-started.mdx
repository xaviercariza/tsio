# Getting Started

This guide creates a minimal typed realtime API with one action and one event.

## Install

```bash
pnpm add @tsio/core zod
```

Then install one transport adapter:

```bash
pnpm add @tsio/socketio socket.io socket.io-client
```

or:

```bash
pnpm add @tsio/ws ws
```

## Define a contract

```ts
import { contract } from '@tsio/core'
import { z } from 'zod'

const MessageSchema = z.object({
  id: z.string(),
  text: z.string(),
})

export const api = contract({
  chat: {
    sendMessage: {
      type: 'action',
      input: z.object({ text: z.string() }),
      response: MessageSchema,
    },
    onMessage: {
      type: 'event',
      data: MessageSchema,
    },
  },
})
```

The contract is the source of truth for action input, action response, event payloads, and nested dot paths.

## Create the server router

```ts
import { createServer } from '@tsio/core'
import { api } from './contract'

type Context = {
  userId: string
}

const tsio = createServer.context<Context>().create(api)

export const router = tsio.router.create(a => ({
  chat: {
    sendMessage: a.chat.sendMessage.handle(({ ctx, input, emit }) => {
      const message = {
        id: crypto.randomUUID(),
        text: input.text,
      }

      emit('chat.onMessage', ctx.userId, message)

      return {
        success: true,
        data: message,
      }
    }),
  },
}))
```

The handler receives typed `ctx`, `input`, `path`, and `emit`.

## Create a typed client

```ts
import { createClient } from '@tsio/core'
import { socketioClient } from '@tsio/socketio/client'
import { io } from 'socket.io-client'
import { api } from './contract'

const socket = io('http://localhost:3000')
const adapter = socketioClient(socket)
const client = createClient(api, adapter)
```

Now the client is fully typed:

```ts
const result = await client.actions.chat.sendMessage({ text: 'Hello' })

client.events.chat.onMessage(message => {
  message.id
  message.text
})
```

## Attach the router

Socket.IO example:

```ts
import { socketio } from '@tsio/socketio/server'

io.on('connection', socket => {
  const adapter = socketio(socket)

  tsio.attach({
    router,
    adapter,
    createContext: () => ({ userId: socket.id }),
  })
})
```

The router is attached per connection so every action call receives a fresh context.
